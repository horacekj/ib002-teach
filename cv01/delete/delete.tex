\begin{procedure}[H]
	\caption{Delete1($\mathit{L,node}$)}
	\Input{seznam $L$ typu $\mathit{List}$, ukazatel $\mathit{node}$ na prvek seznamu, který chceme odstranit }
	\BlankLine
	\If{$\mathit{node} = \mathit{nil}$}{
		\Return{Chyba, byl zadán prázdný ukazatel}
	}
	\If{$\mathit{node.prev} \neq \mathit{nil}$}{
		$\mathit{node.prev.next} \leftarrow \mathit{node.next}$ \;
	}
	\If{$\mathit{node.next} \neq \mathit{nil}$}{
		$\mathit{node.next.prev} \leftarrow \mathit{node.prev}$ \;
	}
	\ffrelease{$\mathit{node}$} \tcp{až po úpravě ukazatelů uvolňujeme prvek z~paměti}
\end{procedure}

\begin{itemize}
	\setlength{\itemsep}{0em}
	\item Skončí hezkou chybou, pokud \textit{L = nil}.
	\item Neumí korektně odstranit první prvek.
	\item Neumí korektně odstranit poslední prvek.
	\item Umí korektně odstranit prvek uprostřed.
	\item Uvolní \textit{node}.
\end{itemize}
