\begin{function}[H]
	\caption{CountingSort($A,n,k$)}
	\Input{pole $A$ délky $n$ s~hodnotami z~intervalu $\langle 1,\dots,k \rangle$}
	\Output{vzestupně seřazené pole $A$ původních prvků}
	\BlankLine

	\For{$i \leftarrow 1$ \KwTo $k$}{ }
	\For{$i \leftarrow 1$ \KwTo $n$}{ }
	\For{$i \leftarrow 2$ \KwTo $k$}{ }
	\For{$i \leftarrow n$ \KwDownTo $1$}{ \\ }

	\Return{$B$}
\end{function}
